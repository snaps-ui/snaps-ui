## Usage

```tsx
import { Field } from '@snaps-ui/react'

;<Field.Root>
  <Field.Label>
    <Field.RequiredIndicator />
  </Field.Label>
  <Input />
  <Field.HelperText />
  <Field.ErrorText />
</Field.Root>
```

## Examples

### Error Text

Pass the **invalid** prop to **Field.Root** and use the **Field.Error** to indicate that the field is invalid.
