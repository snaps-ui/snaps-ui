## Usage

```tsx
import { Table } from '@snaps-ui/react'

;<Table.Root>
  <Table.Header>
    <Table.Row>
      <Table.HeadCell />
    </Table.Row>
  </Table.Header>
  <Table.Body>
    <Table.Row>
      <Table.Cell />
    </Table.Row>
  </Table.Body>
</Table.Root>
```
