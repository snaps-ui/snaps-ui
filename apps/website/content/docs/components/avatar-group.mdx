## Usage

```tsx
import { Avatar, AvatarGroup } from '@snaps-ui/react/avatar'

;<AvatarGroup>
  <Avatar.Root>
    <Avatar.Fallback />
    <Avatar.Image />
  </Avatar.Root>
</AvatarGroup>
```
